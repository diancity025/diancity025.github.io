00189 /**
00190   * @brief  Initialize USART registers according to the specified
00191   *         parameters in USART_InitStruct.
00192   * @note   As some bits in USART configuration registers can only be written when the USART is disabled (USART_CR1_UE bit =0),
00193   *         USART IP should be in disabled state prior calling this function. Otherwise, ERROR result will be returned.
00194   * @note   Baud rate value stored in USART_InitStruct BaudRate field, should be valid (different from 0).
00195   * @param  USARTx USART Instance
00196   * @param  USART_InitStruct pointer to a LL_USART_InitTypeDef structure
00197   *         that contains the configuration information for the specified USART peripheral.
00198   * @retval An ErrorStatus enumeration value:
00199   *          - SUCCESS: USART registers are initialized according to USART_InitStruct content
00200   *          - ERROR: Problem occurred during USART Registers initialization
00201   */
00202 ErrorStatus LL_USART_Init(USART_TypeDef *USARTx, LL_USART_InitTypeDef *USART_InitStruct)
00203 {
00204   ErrorStatus status = ERROR;
00205   uint32_t periphclk = LL_RCC_PERIPH_FREQUENCY_NO;
00206   LL_RCC_ClocksTypeDef rcc_clocks;
00207 
00208   /* Check the parameters */
00209   assert_param(IS_UART_INSTANCE(USARTx));
00210   assert_param(IS_LL_USART_BAUDRATE(USART_InitStruct->BaudRate));
00211   assert_param(IS_LL_USART_DATAWIDTH(USART_InitStruct->DataWidth));
00212   assert_param(IS_LL_USART_STOPBITS(USART_InitStruct->StopBits));
00213   assert_param(IS_LL_USART_PARITY(USART_InitStruct->Parity));
00214   assert_param(IS_LL_USART_DIRECTION(USART_InitStruct->TransferDirection));
00215   assert_param(IS_LL_USART_HWCONTROL(USART_InitStruct->HardwareFlowControl));
00216 #if defined(USART_CR1_OVER8)
00217   assert_param(IS_LL_USART_OVERSAMPLING(USART_InitStruct->OverSampling));
00218 #endif /* USART_OverSampling_Feature */
00219 
00220   /* USART needs to be in disabled state, in order to be able to configure some bits in
00221      CRx registers */
00222   if (LL_USART_IsEnabled(USARTx) == 0U)
00223   {
00224     /*---------------------------- USART CR1 Configuration -----------------------
00225      * Configure USARTx CR1 (USART Word Length, Parity, Mode and Oversampling bits) with parameters:
00226      * - DataWidth:          USART_CR1_M bits according to USART_InitStruct->DataWidth value
00227      * - Parity:             USART_CR1_PCE, USART_CR1_PS bits according to USART_InitStruct->Parity value
00228      * - TransferDirection:  USART_CR1_TE, USART_CR1_RE bits according to USART_InitStruct->TransferDirection value
00229      * - Oversampling:       USART_CR1_OVER8 bit according to USART_InitStruct->OverSampling value.
00230      */
00231 #if defined(USART_CR1_OVER8)
00232     MODIFY_REG(USARTx->CR1,
00233                (USART_CR1_M | USART_CR1_PCE | USART_CR1_PS |
00234                 USART_CR1_TE | USART_CR1_RE | USART_CR1_OVER8),
00235                (USART_InitStruct->DataWidth | USART_InitStruct->Parity |
00236                 USART_InitStruct->TransferDirection | USART_InitStruct->OverSampling));
00237 #else
00238     MODIFY_REG(USARTx->CR1,
00239                (USART_CR1_M | USART_CR1_PCE | USART_CR1_PS |
00240                 USART_CR1_TE | USART_CR1_RE),
00241                (USART_InitStruct->DataWidth | USART_InitStruct->Parity |
00242                 USART_InitStruct->TransferDirection));
00243 #endif /* USART_OverSampling_Feature */
00244 
00245     /*---------------------------- USART CR2 Configuration -----------------------
00246      * Configure USARTx CR2 (Stop bits) with parameters:
00247      * - Stop Bits:          USART_CR2_STOP bits according to USART_InitStruct->StopBits value.
00248      * - CLKEN, CPOL, CPHA and LBCL bits are to be configured using LL_USART_ClockInit().
00249      */
00250     LL_USART_SetStopBitsLength(USARTx, USART_InitStruct->StopBits);
00251 
00252     /*---------------------------- USART CR3 Configuration -----------------------
00253      * Configure USARTx CR3 (Hardware Flow Control) with parameters:
00254      * - HardwareFlowControl: USART_CR3_RTSE, USART_CR3_CTSE bits according to USART_InitStruct->HardwareFlowControl value.
00255      */
00256     LL_USART_SetHWFlowCtrl(USARTx, USART_InitStruct->HardwareFlowControl);
00257 
00258     /*---------------------------- USART BRR Configuration -----------------------
00259      * Retrieve Clock frequency used for USART Peripheral
00260      */
00261     LL_RCC_GetSystemClocksFreq(&rcc_clocks);
00262     if (USARTx == USART1)
00263     {
00264       periphclk = rcc_clocks.PCLK2_Frequency;
00265     }
00266     else if (USARTx == USART2)
00267     {
00268       periphclk = rcc_clocks.PCLK1_Frequency;
00269     }
00270 #if defined(USART3)
00271     else if (USARTx == USART3)
00272     {
00273       periphclk = rcc_clocks.PCLK1_Frequency;
00274     }
00275 #endif /* USART3 */
00276 #if defined(UART4)
00277     else if (USARTx == UART4)
00278     {
00279       periphclk = rcc_clocks.PCLK1_Frequency;
00280     }
00281 #endif /* UART4 */
00282 #if defined(UART5)
00283     else if (USARTx == UART5)
00284     {
00285       periphclk = rcc_clocks.PCLK1_Frequency;
00286     }
00287 #endif /* UART5 */
00288     else
00289     {
00290       /* Nothing to do, as error code is already assigned to ERROR value */
00291     }
00292 
00293     /* Configure the USART Baud Rate :
00294        - valid baud rate value (different from 0) is required
00295        - Peripheral clock as returned by RCC service, should be valid (different from 0).
00296     */
00297     if ((periphclk != LL_RCC_PERIPH_FREQUENCY_NO)
00298         && (USART_InitStruct->BaudRate != 0U))
00299     {
00300       status = SUCCESS;
00301 #if defined(USART_CR1_OVER8)
00302       LL_USART_SetBaudRate(USARTx,
00303                            periphclk,
00304                            USART_InitStruct->OverSampling,
00305                            USART_InitStruct->BaudRate);
00306 #else
00307       LL_USART_SetBaudRate(USARTx,
00308                            periphclk,
00309                            USART_InitStruct->BaudRate);
00310 #endif /* USART_OverSampling_Feature */
00311 
00312       /* Check BRR is greater than or equal to 16d */
00313       assert_param(IS_LL_USART_BRR_MIN(USARTx->BRR));
00314 
00315       /* Check BRR is greater than or equal to 16d */
00316       assert_param(IS_LL_USART_BRR_MAX(USARTx->BRR));
00317     }
00318   }
00319   /* Endif (=> USART not in Disabled state => return ERROR) */
00320 
00321   return (status);
00322 }

00345 /**
00346   * @brief  Initialize USART Clock related settings according to the
00347   *         specified parameters in the USART_ClockInitStruct.
00348   * @note   As some bits in USART configuration registers can only be written when the USART is disabled (USART_CR1_UE bit =0),
00349   *         USART IP should be in disabled state prior calling this function. Otherwise, ERROR result will be returned.
00350   * @param  USARTx USART Instance
00351   * @param  USART_ClockInitStruct Pointer to a @ref LL_USART_ClockInitTypeDef structure
00352   *         that contains the Clock configuration information for the specified USART peripheral.
00353   * @retval An ErrorStatus enumeration value:
00354   *          - SUCCESS: USART registers related to Clock settings are initialized according to USART_ClockInitStruct content
00355   *          - ERROR: Problem occurred during USART Registers initialization
00356   */
00357 ErrorStatus LL_USART_ClockInit(USART_TypeDef *USARTx, LL_USART_ClockInitTypeDef *USART_ClockInitStruct)
00358 {
00359   ErrorStatus status = SUCCESS;
00360 
00361   /* Check USART Instance and Clock signal output parameters */
00362   assert_param(IS_UART_INSTANCE(USARTx));
00363   assert_param(IS_LL_USART_CLOCKOUTPUT(USART_ClockInitStruct->ClockOutput));
00364 
00365   /* USART needs to be in disabled state, in order to be able to configure some bits in
00366      CRx registers */
00367   if (LL_USART_IsEnabled(USARTx) == 0U)
00368   {
00369     /*---------------------------- USART CR2 Configuration -----------------------*/
00370     /* If Clock signal has to be output */
00371     if (USART_ClockInitStruct->ClockOutput == LL_USART_CLOCK_DISABLE)
00372     {
00373       /* Deactivate Clock signal delivery :
00374        * - Disable Clock Output:        USART_CR2_CLKEN cleared
00375        */
00376       LL_USART_DisableSCLKOutput(USARTx);
00377     }
00378     else
00379     {
00380       /* Ensure USART instance is USART capable */
00381       assert_param(IS_USART_INSTANCE(USARTx));
00382 
00383       /* Check clock related parameters */
00384       assert_param(IS_LL_USART_CLOCKPOLARITY(USART_ClockInitStruct->ClockPolarity));
00385       assert_param(IS_LL_USART_CLOCKPHASE(USART_ClockInitStruct->ClockPhase));
00386       assert_param(IS_LL_USART_LASTBITCLKOUTPUT(USART_ClockInitStruct->LastBitClockPulse));
00387 
00388       /*---------------------------- USART CR2 Configuration -----------------------
00389        * Configure USARTx CR2 (Clock signal related bits) with parameters:
00390        * - Enable Clock Output:         USART_CR2_CLKEN set
00391        * - Clock Polarity:              USART_CR2_CPOL bit according to USART_ClockInitStruct->ClockPolarity value
00392        * - Clock Phase:                 USART_CR2_CPHA bit according to USART_ClockInitStruct->ClockPhase value
00393        * - Last Bit Clock Pulse Output: USART_CR2_LBCL bit according to USART_ClockInitStruct->LastBitClockPulse value.
00394        */
00395       MODIFY_REG(USARTx->CR2,
00396                  USART_CR2_CLKEN | USART_CR2_CPHA | USART_CR2_CPOL | USART_CR2_LBCL,
00397                  USART_CR2_CLKEN | USART_ClockInitStruct->ClockPolarity |
00398                  USART_ClockInitStruct->ClockPhase | USART_ClockInitStruct->LastBitClockPulse);
00399     }
00400   }
00401   /* Else (USART not in Disabled state => return ERROR */
00402   else
00403   {
00404     status = ERROR;
00405   }
00406 
00407   return (status);
00408 }
