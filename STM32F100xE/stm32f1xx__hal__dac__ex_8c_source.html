00200   * @retval HAL status
00201   */
00202 HAL_StatusTypeDef HAL_DACEx_DualSetValue(DAC_HandleTypeDef *hdac, uint32_t Alignment, uint32_t Data1, uint32_t Data2)
00203 {
00204   uint32_t data;
00205   uint32_t tmp;
00206 
00207   /* Check the parameters */
00208   assert_param(IS_DAC_ALIGN(Alignment));
00209   assert_param(IS_DAC_DATA(Data1));
00210   assert_param(IS_DAC_DATA(Data2));
00211 
00212   /* Calculate and set dual DAC data holding register value */
00213   if (Alignment == DAC_ALIGN_8B_R)
00214   {
00215     data = ((uint32_t)Data2 << 8U) | Data1;
00216   }
00217   else
00218   {
00219     data = ((uint32_t)Data2 << 16U) | Data1;
00220   }
00221 
00222   tmp = (uint32_t)hdac->Instance;
00223   tmp += DAC_DHR12RD_ALIGNMENT(Alignment);
00224 
00225   /* Set the dual DAC selected data holding register */
00226   *(__IO uint32_t *)tmp = data;
00227 
00228   /* Return function status */
00229   return HAL_OK;
