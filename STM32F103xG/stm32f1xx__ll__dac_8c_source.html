00148 /**
00149   * @brief  Initialize some features of DAC channel.
00150   * @note   @ref LL_DAC_Init() aims to ease basic configuration of a DAC channel.
00151   *         Leaving it ready to be enabled and output:
00152   *         a level by calling one of
00153   *           @ref LL_DAC_ConvertData12RightAligned
00154   *           @ref LL_DAC_ConvertData12LeftAligned
00155   *           @ref LL_DAC_ConvertData8RightAligned
00156   *         or one of the supported autogenerated wave.
00157   * @note   This function allows configuration of:
00158   *          - Output mode
00159   *          - Trigger
00160   *          - Wave generation
00161   * @note   The setting of these parameters by function @ref LL_DAC_Init()
00162   *         is conditioned to DAC state:
00163   *         DAC channel must be disabled.
00164   * @param  DACx DAC instance
00165   * @param  DAC_Channel This parameter can be one of the following values:
00166   *         @arg @ref LL_DAC_CHANNEL_1
00167   *         @arg @ref LL_DAC_CHANNEL_2
00168   * @param  DAC_InitStruct Pointer to a @ref LL_DAC_InitTypeDef structure
00169   * @retval An ErrorStatus enumeration value:
00170   *          - SUCCESS: DAC registers are initialized
00171   *          - ERROR: DAC registers are not initialized
00172   */
00173 ErrorStatus LL_DAC_Init(DAC_TypeDef *DACx, uint32_t DAC_Channel, LL_DAC_InitTypeDef *DAC_InitStruct)
00174 {
00175   ErrorStatus status = SUCCESS;
00176 
00177   /* Check the parameters */
00178   assert_param(IS_DAC_ALL_INSTANCE(DACx));
00179   assert_param(IS_LL_DAC_CHANNEL(DAC_Channel));
00180   assert_param(IS_LL_DAC_TRIGGER_SOURCE(DAC_InitStruct->TriggerSource));
00181   assert_param(IS_LL_DAC_OUTPUT_BUFFER(DAC_InitStruct->OutputBuffer));
00182   assert_param(IS_LL_DAC_WAVE_AUTO_GENER_MODE(DAC_InitStruct->WaveAutoGeneration));
00183   if (DAC_InitStruct->WaveAutoGeneration != LL_DAC_WAVE_AUTO_GENERATION_NONE)
00184   {
00185     assert_param(IS_LL_DAC_WAVE_AUTO_GENER_CONFIG(DAC_InitStruct->WaveAutoGeneration,
00186                                                   DAC_InitStruct->WaveAutoGenerationConfig));
00187   }
00188 
00189   /* Note: Hardware constraint (refer to description of this function)        */
00190   /*       DAC instance must be disabled.                                     */
00191   if (LL_DAC_IsEnabled(DACx, DAC_Channel) == 0U)
00192   {
00193     /* Configuration of DAC channel:                                          */
00194     /*  - TriggerSource                                                       */
00195     /*  - WaveAutoGeneration                                                  */
00196     /*  - OutputBuffer                                                        */
00197     /*  - OutputMode                                                          */
00198     if (DAC_InitStruct->WaveAutoGeneration != LL_DAC_WAVE_AUTO_GENERATION_NONE)
00199     {
00200       MODIFY_REG(DACx->CR,
00201                  (DAC_CR_TSEL1
00202                   | DAC_CR_WAVE1
00203                   | DAC_CR_MAMP1
00204                   | DAC_CR_BOFF1
00205                  ) << (DAC_Channel & DAC_CR_CHX_BITOFFSET_MASK)
00206                  ,
00207                  (DAC_InitStruct->TriggerSource
00208                   | DAC_InitStruct->WaveAutoGeneration
00209                   | DAC_InitStruct->WaveAutoGenerationConfig
00210                   | DAC_InitStruct->OutputBuffer
00211                  ) << (DAC_Channel & DAC_CR_CHX_BITOFFSET_MASK)
00212                 );
00213     }
00214     else
00215     {
00216       MODIFY_REG(DACx->CR,
00217                  (DAC_CR_TSEL1
00218                   | DAC_CR_WAVE1
00219                   | DAC_CR_BOFF1
00220                  ) << (DAC_Channel & DAC_CR_CHX_BITOFFSET_MASK)
00221                  ,
00222                  (DAC_InitStruct->TriggerSource
00223                   | LL_DAC_WAVE_AUTO_GENERATION_NONE
00224                   | DAC_InitStruct->OutputBuffer
00225                  ) << (DAC_Channel & DAC_CR_CHX_BITOFFSET_MASK)
00226                 );
00227     }
00228   }
00229   else
00230   {
00231     /* Initialization error: DAC instance is not disabled.                    */
00232     status = ERROR;
00233   }
00234   return status;
00235 }
