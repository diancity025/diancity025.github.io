00044 /** 
00045   * @brief  Structure definition of ADC and regular group initialization 
00046   * @note   Parameters of this structure are shared within 2 scopes:
00047   *          - Scope entire ADC (affects regular and injected groups): DataAlign, ScanConvMode.
00048   *          - Scope regular group: ContinuousConvMode, NbrOfConversion, DiscontinuousConvMode, NbrOfDiscConversion, ExternalTrigConvEdge, ExternalTrigConv.
00049   * @note   The setting of these parameters with function HAL_ADC_Init() is conditioned to ADC state.
00050   *         ADC can be either disabled or enabled without conversion on going on regular group.
00051   */
00052 typedef struct
00053 {
00054   uint32_t DataAlign;             /*!< Specifies ADC data alignment to right (MSB on register bit 11 and LSB on register bit 0) (default setting)
00055                                        or to left (if regular group: MSB on register bit 15 and LSB on register bit 4, if injected group (MSB kept as signed value due to potential negative value after offset application): MSB on register bit 14 and LSB on register bit 3).
00056                                        This parameter can be a value of @ref ADC_Data_align */
00057   uint32_t ScanConvMode;          /*!< Configures the sequencer of regular and injected groups.
00058                                        This parameter can be associated to parameter 'DiscontinuousConvMode' to have main sequence subdivided in successive parts.
00059                                        If disabled: Conversion is performed in single mode (one channel converted, the one defined in rank 1).
00060                                                     Parameters 'NbrOfConversion' and 'InjectedNbrOfConversion' are discarded (equivalent to set to 1).
00061                                        If enabled:  Conversions are performed in sequence mode (multiple ranks defined by 'NbrOfConversion'/'InjectedNbrOfConversion' and each channel rank).
00062                                                     Scan direction is upward: from rank1 to rank 'n'.
00063                                        This parameter can be a value of @ref ADC_Scan_mode
00064                                        Note: For regular group, this parameter should be enabled in conversion either by polling (HAL_ADC_Start with Discontinuous mode and NbrOfDiscConversion=1)
00065                                              or by DMA (HAL_ADC_Start_DMA), but not by interruption (HAL_ADC_Start_IT): in scan mode, interruption is triggered only on the
00066                                              the last conversion of the sequence. All previous conversions would be overwritten by the last one.
00067                                              Injected group used with scan mode has not this constraint: each rank has its own result register, no data is overwritten. */
00068   uint32_t ContinuousConvMode;    /*!< Specifies whether the conversion is performed in single mode (one conversion) or continuous mode for regular group,
00069                                        after the selected trigger occurred (software start or external trigger).
00070                                        This parameter can be set to ENABLE or DISABLE. */
00071   uint32_t NbrOfConversion;       /*!< Specifies the number of ranks that will be converted within the regular group sequencer.
00072                                        To use regular group sequencer and convert several ranks, parameter 'ScanConvMode' must be enabled.
00073                                        This parameter must be a number between Min_Data = 1 and Max_Data = 16. */
00074   uint32_t DiscontinuousConvMode; /*!< Specifies whether the conversions sequence of regular group is performed in Complete-sequence/Discontinuous-sequence (main sequence subdivided in successive parts).
00075                                        Discontinuous mode is used only if sequencer is enabled (parameter 'ScanConvMode'). If sequencer is disabled, this parameter is discarded.
00076                                        Discontinuous mode can be enabled only if continuous mode is disabled. If continuous mode is enabled, this parameter setting is discarded.
00077                                        This parameter can be set to ENABLE or DISABLE. */
00078   uint32_t NbrOfDiscConversion;   /*!< Specifies the number of discontinuous conversions in which the  main sequence of regular group (parameter NbrOfConversion) will be subdivided.
00079                                        If parameter 'DiscontinuousConvMode' is disabled, this parameter is discarded.
00080                                        This parameter must be a number between Min_Data = 1 and Max_Data = 8. */
00081   uint32_t ExternalTrigConv;      /*!< Selects the external event used to trigger the conversion start of regular group.
00082                                        If set to ADC_SOFTWARE_START, external triggers are disabled.
00083                                        If set to external trigger source, triggering is on event rising edge.
00084                                        This parameter can be a value of @ref ADC_External_trigger_source_Regular */
00085 }ADC_InitTypeDef;
