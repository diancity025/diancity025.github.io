00871         return HAL_ERROR;

01080 {

01082   const uint8_t aPLLMULFactorTable[14] = {0, 0, 4, 5, 6, 7, 8, 9, 0, 0, 0, 0, 0, 13};

01088 #else
01089   const uint8_t aPredivFactorTable[2] = {1, 2};
01090 #endif /*RCC_CFGR2_PREDIV1*/
01091 
01092 #endif
01093   uint32_t tmpreg = 0U, prediv = 0U, pllclk = 0U, pllmul = 0U;
01094   uint32_t sysclockfreq = 0U;
01095 #if defined(RCC_CFGR2_PREDIV1SRC)

01097 #endif /*RCC_CFGR2_PREDIV1SRC*/

01103   {
01104     case RCC_SYSCLKSOURCE_STATUS_HSE:  /* HSE used as system clock */
01105     {
01106       sysclockfreq = HSE_VALUE;
01107       break;
01108     }
01109     case RCC_SYSCLKSOURCE_STATUS_PLLCLK:  /* PLL used as system clock */
01110     {
01111       pllmul = aPLLMULFactorTable[(uint32_t)(tmpreg & RCC_CFGR_PLLMULL) >> RCC_CFGR_PLLMULL_Pos];

01113       {
01114 #if defined(RCC_CFGR2_PREDIV1)
01115         prediv = aPredivFactorTable[(uint32_t)(RCC->CFGR2 & RCC_CFGR2_PREDIV1) >> RCC_CFGR2_PREDIV1_Pos];
01116 #else
01117         prediv = aPredivFactorTable[(uint32_t)(RCC->CFGR & RCC_CFGR_PLLXTPRE) >> RCC_CFGR_PLLXTPRE_Pos];
01118 #endif /*RCC_CFGR2_PREDIV1*/
01119 #if defined(RCC_CFGR2_PREDIV1SRC)
01120 
01121         if(HAL_IS_BIT_SET(RCC->CFGR2, RCC_CFGR2_PREDIV1SRC))
01122         {
01123           /* PLL2 selected as Prediv1 source */
01124           /* PLLCLK = PLL2CLK / PREDIV1 * PLLMUL with PLL2CLK = HSE/PREDIV2 * PLL2MUL */
01125           prediv2 = ((RCC->CFGR2 & RCC_CFGR2_PREDIV2) >> RCC_CFGR2_PREDIV2_Pos) + 1;
01126           pll2mul = ((RCC->CFGR2 & RCC_CFGR2_PLL2MUL) >> RCC_CFGR2_PLL2MUL_Pos) + 2;
01127           pllclk = (uint32_t)(((uint64_t)HSE_VALUE * (uint64_t)pll2mul * (uint64_t)pllmul) / ((uint64_t)prediv2 * (uint64_t)prediv));
01128         }
01129         else
01130         {
01131           /* HSE used as PLL clock source : PLLCLK = HSE/PREDIV1 * PLLMUL */
01132           pllclk = (uint32_t)((HSE_VALUE * pllmul) / prediv);
01133         }
01134 
01135         /* If PLLMUL was set to 13 means that it was to cover the case PLLMUL 6.5 (avoid using float) */
01136         /* In this case need to divide pllclk by 2 */
01137         if (pllmul == aPLLMULFactorTable[(uint32_t)(RCC_CFGR_PLLMULL6_5) >> RCC_CFGR_PLLMULL_Pos])
01138         {
01139             pllclk = pllclk / 2;
01140         }
01141 #else

01144 #endif /*RCC_CFGR2_PREDIV1SRC*/
01145       }
01146       else
01147       {
01148         /* HSI used as PLL clock source : PLLCLK = HSI/2 * PLLMUL */
01149         pllclk = (uint32_t)((HSI_VALUE >> 1) * pllmul);
01150       }
01151       sysclockfreq = pllclk;
01152       break;
01153     }
01154     case RCC_SYSCLKSOURCE_STATUS_HSI:  /* HSI used as system clock source */
01155     default: /* HSI used as system clock */

01157       sysclockfreq = HSI_VALUE;
01158       break;
01159     }
01160   }
01161   return sysclockfreq;
