01408   * @param  hi2s pointer to a I2S_HandleTypeDef structure that contains
01409   *         the configuration information for I2S module
01410   * @retval None
01411   */
01412 void HAL_I2S_IRQHandler(I2S_HandleTypeDef *hi2s)
01413 {
01414   uint32_t itsource = hi2s->Instance->CR2;
01415   uint32_t itflag   = hi2s->Instance->SR;
01416 
01417   /* I2S in mode Receiver ------------------------------------------------*/
01418   if ((I2S_CHECK_FLAG(itflag, I2S_FLAG_OVR) == RESET) &&
01419       (I2S_CHECK_FLAG(itflag, I2S_FLAG_RXNE) != RESET) && (I2S_CHECK_IT_SOURCE(itsource, I2S_IT_RXNE) != RESET))
01420   {
01421     I2S_Receive_IT(hi2s);
01422     return;
01423   }
01424 
01425   /* I2S in mode Tramitter -----------------------------------------------*/
01426   if ((I2S_CHECK_FLAG(itflag, I2S_FLAG_TXE) != RESET) && (I2S_CHECK_IT_SOURCE(itsource, I2S_IT_TXE) != RESET))
01427   {
01428     I2S_Transmit_IT(hi2s);
01429     return;
01430   }
01431

01433   if (I2S_CHECK_IT_SOURCE(itsource, I2S_IT_ERR) != RESET)

01436     if (I2S_CHECK_FLAG(itflag, I2S_FLAG_OVR) != RESET)
01437     {
01438       /* Disable RXNE and ERR interrupt */
01439       __HAL_I2S_DISABLE_IT(hi2s, (I2S_IT_RXNE | I2S_IT_ERR));
01440 
01441       /* Set the error code and execute error callback*/
01442       SET_BIT(hi2s->ErrorCode, HAL_I2S_ERROR_OVR);
01443     }
01444 
01445     /* I2S Underrun error interrupt occurred --------------------------------*/

01447     {
01448       /* Disable TXE and ERR interrupt */
01449       __HAL_I2S_DISABLE_IT(hi2s, (I2S_IT_TXE | I2S_IT_ERR));
01450 
01451       /* Set the error code and execute error callback*/
01452       SET_BIT(hi2s->ErrorCode, HAL_I2S_ERROR_UDR);
01453     }
01454 
01455     /* Set the I2S State ready */
01456     hi2s->State = HAL_I2S_STATE_READY;
01457 
01458     /* Call user error callback */
01459 #if (USE_HAL_I2S_REGISTER_CALLBACKS == 1U)
01460     hi2s->ErrorCallback(hi2s);
