00176   * @brief  This function provides accurate delay (in milliseconds) based

00178   * @note   When a RTOS is used, it is recommended to avoid using blocking delay

00186 {
00187   __IO uint32_t  tmp = SysTick->CTRL;  /* Clear the COUNTFLAG first */
00188   /* Add this code to indicate that local variable is not used */
00189   ((void)tmp);
00190

00192   if (Delay < LL_MAX_DELAY)
00193   {
00194     Delay++;
00195   }
00196 
00197   while (Delay)
00198   {
00199     if ((SysTick->CTRL & SysTick_CTRL_COUNTFLAG_Msk) != 0U)
00200     {
00201       Delay--;
00202     }
00203   }
