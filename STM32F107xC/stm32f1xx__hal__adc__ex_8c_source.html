00410   /*    from ADC conversion time (selected sampling time + conversion time of */
00411   /*    12.5 ADC clock cycles) and APB2/ADC clock prescalers (depending on    */

00416   {
00417     /* Wait until End of Conversion flag is raised */
00418     while(HAL_IS_BIT_CLR(hadc->Instance->SR, ADC_FLAG_JEOC))
00419     {
00420       /* Check if timeout is disabled (set to infinite wait) */

00426           SET_BIT(hadc->State, HAL_ADC_STATE_TIMEOUT);
00427           
00428           /* Process unlocked */
00429           __HAL_UNLOCK(hadc);
00430           
00431           return HAL_TIMEOUT;
00432         }
00433       }
00434     }
00435   }

00452         {
00453           /* Update ADC state machine to timeout */
00454           SET_BIT(hadc->State, HAL_ADC_STATE_TIMEOUT);
00455 
00456           /* Process unlocked */
00457           __HAL_UNLOCK(hadc);
00458           
00459           return HAL_TIMEOUT;

00465

00486     {
00487       SET_BIT(hadc->State, HAL_ADC_STATE_READY);
00488     }
00489   }
00490

00984   *         linked to the injected group.

00986   *         This function initializes injected group, following calls to this

00988   *         "ADC_InjectionConfTypeDef" on the fly, without reseting the ADC.
00989   *         The setting of these parameters is conditioned to ADC state: 
00990   *         this function must be called when ADC is not under conversion.
00991   * @param  hadc: ADC handle
00992   * @param  sConfigInjected: Structure of ADC injected group and ADC channel for
00993   *         injected group.
00994   * @retval None
00995   */
00996 HAL_StatusTypeDef HAL_ADCEx_InjectedConfigChannel(ADC_HandleTypeDef* hadc, ADC_InjectionConfTypeDef* sConfigInjected)
00997 {
00998   HAL_StatusTypeDef tmp_hal_status = HAL_OK;
00999   __IO uint32_t wait_loop_index = 0U;
01000   
01001   /* Check the parameters */
01002   assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
01003   assert_param(IS_ADC_CHANNEL(sConfigInjected->InjectedChannel));
01004   assert_param(IS_ADC_SAMPLE_TIME(sConfigInjected->InjectedSamplingTime));
01005   assert_param(IS_FUNCTIONAL_STATE(sConfigInjected->AutoInjectedConv));
01006   assert_param(IS_ADC_EXTTRIGINJEC(sConfigInjected->ExternalTrigInjecConv));
01007   assert_param(IS_ADC_RANGE(sConfigInjected->InjectedOffset));
01008   
01009   if(hadc->Init.ScanConvMode != ADC_SCAN_DISABLE)
01010   {
01011     assert_param(IS_ADC_INJECTED_RANK(sConfigInjected->InjectedRank));
01012     assert_param(IS_ADC_INJECTED_NB_CONV(sConfigInjected->InjectedNbrOfConversion));
01013     assert_param(IS_FUNCTIONAL_STATE(sConfigInjected->InjectedDiscontinuousConvMode));
01014   }
01015   
01016   /* Process locked */
01017   __HAL_LOCK(hadc);
01018   
01019   /* Configuration of injected group sequencer:                               */
01020   /* - if scan mode is disabled, injected channels sequence length is set to  */
01021   /*   0x00: 1 channel converted (channel on regular rank 1)                  */
01022   /*   Parameter "InjectedNbrOfConversion" is discarded.                      */
01023   /*   Note: Scan mode is present by hardware on this device and, if          */
01024   /*   disabled, discards automatically nb of conversions. Anyway, nb of      */
01025   /*   conversions is forced to 0x00 for alignment over all STM32 devices.    */
01026   /* - if scan mode is enabled, injected channels sequence length is set to   */
01027   /*   parameter "InjectedNbrOfConversion".                                   */
01028   if (hadc->Init.ScanConvMode == ADC_SCAN_DISABLE)
01029   {
01030     if (sConfigInjected->InjectedRank == ADC_INJECTED_RANK_1)
01031     {
01032       /* Clear the old SQx bits for all injected ranks */
01033       MODIFY_REG(hadc->Instance->JSQR                             ,
01034                  ADC_JSQR_JL   |
01035                  ADC_JSQR_JSQ4 |
01036                  ADC_JSQR_JSQ3 |
01037                  ADC_JSQR_JSQ2 |
01038                  ADC_JSQR_JSQ1                                    ,
01039                  ADC_JSQR_RK_JL(sConfigInjected->InjectedChannel,
01040                                   ADC_INJECTED_RANK_1,
01041                                   0x01U));
01042     }
01043     /* If another injected rank than rank1 was intended to be set, and could  */
01044     /* not due to ScanConvMode disabled, error is reported.                   */
01045     else
01046     {
01047       /* Update ADC state machine to error */
01048       SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);
01049       
01050       tmp_hal_status = HAL_ERROR;
01051     }
01052   }
01053   else
01054   {
01055     /* Since injected channels rank conv. order depends on total number of   */
01056     /* injected conversions, selected rank must be below or equal to total   */
01057     /* number of injected conversions to be updated.                         */
01058     if (sConfigInjected->InjectedRank <= sConfigInjected->InjectedNbrOfConversion)
01059     {
01060       /* Clear the old SQx bits for the selected rank */
01061       /* Set the SQx bits for the selected rank */
01062       MODIFY_REG(hadc->Instance->JSQR                                         ,
01063                  
01064                  ADC_JSQR_JL                                               |
01065                  ADC_JSQR_RK_JL(ADC_JSQR_JSQ1,                         
01066                                   sConfigInjected->InjectedRank,         
01067                                   sConfigInjected->InjectedNbrOfConversion)   ,
01068                  
01069                  ADC_JSQR_JL_SHIFT(sConfigInjected->InjectedNbrOfConversion) |
01070                  ADC_JSQR_RK_JL(sConfigInjected->InjectedChannel,      
01071                                   sConfigInjected->InjectedRank,         
01072                                   sConfigInjected->InjectedNbrOfConversion)    );
01073     }
01074     else
01075     {
01076       /* Clear the old SQx bits for the selected rank */
01077       MODIFY_REG(hadc->Instance->JSQR                                       ,
01078                  
01079                  ADC_JSQR_JL                                               |
01080                  ADC_JSQR_RK_JL(ADC_JSQR_JSQ1,                         
01081                                   sConfigInjected->InjectedRank,         
01082                                   sConfigInjected->InjectedNbrOfConversion) ,
01083                  
01084                  0x00000000U);
01085     }
01086   } 
01087     
01088   /* Configuration of injected group                                          */
01089   /* Parameters update conditioned to ADC state:                              */
01090   /* Parameters that can be updated only when ADC is disabled:                */
01091   /*  - external trigger to start conversion                                  */
01092   /* Parameters update not conditioned to ADC state:                          */
01093   /*  - Automatic injected conversion                                         */
01094   /*  - Injected discontinuous mode                                           */
01095   /* Note: In case of ADC already enabled, caution to not launch an unwanted  */
01096   /*       conversion while modifying register CR2 by writing 1 to bit ADON.  */
01097   if (ADC_IS_ENABLE(hadc) == RESET)
01098   {    
01099     MODIFY_REG(hadc->Instance->CR2                                           ,
01100                ADC_CR2_JEXTSEL |
01101                ADC_CR2_ADON                                                  ,
01102                ADC_CFGR_JEXTSEL(hadc, sConfigInjected->ExternalTrigInjecConv) );
01103   }
01104   
01105   
01106   /* Configuration of injected group                                          */
01107   /*  - Automatic injected conversion                                         */
01108   /*  - Injected discontinuous mode                                           */
01109   
01110     /* Automatic injected conversion can be enabled if injected group         */
01111     /* external triggers are disabled.                                        */
01112     if (sConfigInjected->AutoInjectedConv == ENABLE)
01113     {
01114       if (sConfigInjected->ExternalTrigInjecConv == ADC_INJECTED_SOFTWARE_START)
01115       {
01116         SET_BIT(hadc->Instance->CR1, ADC_CR1_JAUTO);
01117       }
01118       else
01119       {
01120         /* Update ADC state machine to error */
01121         SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);
01122         
01123         tmp_hal_status = HAL_ERROR;
01124       }
01125     }
01126     
01127     /* Injected discontinuous can be enabled only if auto-injected mode is    */
01128     /* disabled.                                                              */  
01129     if (sConfigInjected->InjectedDiscontinuousConvMode == ENABLE)
01130     {
01131       if (sConfigInjected->AutoInjectedConv == DISABLE)
01132       {
01133         SET_BIT(hadc->Instance->CR1, ADC_CR1_JDISCEN);
01134       } 
01135       else
01136       {
01137         /* Update ADC state machine to error */
01138         SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);
01139         
01140         tmp_hal_status = HAL_ERROR;
01141       }
01142     }
01143 
01144 
01145   /* InjectedChannel sampling time configuration */
01146   /* For channels 10 to 17 */
01147   if (sConfigInjected->InjectedChannel >= ADC_CHANNEL_10)
01148   {
01149     MODIFY_REG(hadc->Instance->SMPR1                                                             ,
01150                ADC_SMPR1(ADC_SMPR1_SMP10, sConfigInjected->InjectedChannel)                      ,
01151                ADC_SMPR1(sConfigInjected->InjectedSamplingTime, sConfigInjected->InjectedChannel) );
01152   }
01153   else /* For channels 0 to 9 */
01154   {
01155     MODIFY_REG(hadc->Instance->SMPR2                                                             ,
01156                ADC_SMPR2(ADC_SMPR2_SMP0, sConfigInjected->InjectedChannel)                       ,
01157                ADC_SMPR2(sConfigInjected->InjectedSamplingTime, sConfigInjected->InjectedChannel) );
01158   }
01159   
01160   /* If ADC1 InjectedChannel_16 or InjectedChannel_17 is selected, enable Temperature sensor  */
01161   /* and VREFINT measurement path.                                            */
01162   if ((sConfigInjected->InjectedChannel == ADC_CHANNEL_TEMPSENSOR) ||
01163       (sConfigInjected->InjectedChannel == ADC_CHANNEL_VREFINT)      )
01164   {
01165     SET_BIT(hadc->Instance->CR2, ADC_CR2_TSVREFE);
01166   }
01167   
01168   
01169   /* Configure the offset: offset enable/disable, InjectedChannel, offset value */
01170   switch(sConfigInjected->InjectedRank)
01171   {
01172     case 1:
01173       /* Set injected channel 1 offset */
01174       MODIFY_REG(hadc->Instance->JOFR1,
01175                  ADC_JOFR1_JOFFSET1,
01176                  sConfigInjected->InjectedOffset);
01177       break;
01178     case 2:
01179       /* Set injected channel 2 offset */
01180       MODIFY_REG(hadc->Instance->JOFR2,
01181                  ADC_JOFR2_JOFFSET2,
01182                  sConfigInjected->InjectedOffset);
01183       break;
01184     case 3:
01185       /* Set injected channel 3 offset */
01186       MODIFY_REG(hadc->Instance->JOFR3,
01187                  ADC_JOFR3_JOFFSET3,
01188                  sConfigInjected->InjectedOffset);
01189       break;
01190     case 4:
01191     default:
01192       MODIFY_REG(hadc->Instance->JOFR4,
01193                  ADC_JOFR4_JOFFSET4,
01194                  sConfigInjected->InjectedOffset);
01195       break;
01196   }
01197   
01198   /* If ADC1 Channel_16 or Channel_17 is selected, enable Temperature sensor  */
01199   /* and VREFINT measurement path.                                            */
01200   if ((sConfigInjected->InjectedChannel == ADC_CHANNEL_TEMPSENSOR) ||
01201       (sConfigInjected->InjectedChannel == ADC_CHANNEL_VREFINT)      )
01202   {
01203     /* For STM32F1 devices with several ADC: Only ADC1 can access internal    */
01204     /* measurement channels (VrefInt/TempSensor). If these channels are       */
01205     /* intended to be set on other ADC instances, an error is reported.       */
01206     if (hadc->Instance == ADC1)
01207     {
01208       if (READ_BIT(hadc->Instance->CR2, ADC_CR2_TSVREFE) == RESET)
01209       {
01210         SET_BIT(hadc->Instance->CR2, ADC_CR2_TSVREFE);
01211         
01212         if ((sConfigInjected->InjectedChannel == ADC_CHANNEL_TEMPSENSOR))
01213         {
01214           /* Delay for temperature sensor stabilization time */
01215           /* Compute number of CPU cycles to wait for */
01216           wait_loop_index = (ADC_TEMPSENSOR_DELAY_US * (SystemCoreClock / 1000000U));
01217           while(wait_loop_index != 0U)
01218           {
01219             wait_loop_index--;
01220           }
01221         }
01222       }
01223     }
01224     else
01225     {
01226       /* Update ADC state machine to error */
01227       SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);
01228       
01229       tmp_hal_status = HAL_ERROR;
01230     }
01231   }
01232   
01233   /* Process unlocked */
01234   __HAL_UNLOCK(hadc);
01235   
01236   /* Return function status */
01237   return tmp_hal_status;
01238 }
