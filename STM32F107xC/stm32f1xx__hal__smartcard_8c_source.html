00020     (#) Declare a SMARTCARD_HandleTypeDef handle structure.
00021     (#) Initialize the SMARTCARD low level resources by implementing the HAL_SMARTCARD_MspInit() API:
00022         (##) Enable the interface clock of the USARTx associated to the SMARTCARD.
00023         (##) SMARTCARD pins configuration:
00024             (+++) Enable the clock for the SMARTCARD GPIOs.
00025             (+++) Configure SMARTCARD pins as alternate function pull-up.
00026         (##) NVIC configuration if you need to use interrupt process (HAL_SMARTCARD_Transmit_IT()
00027              and HAL_SMARTCARD_Receive_IT() APIs):
00028             (+++) Configure the USARTx interrupt priority.
00029             (+++) Enable the NVIC USART IRQ handle.

00031              and HAL_SMARTCARD_Receive_DMA() APIs):
00032             (+++) Declare a DMA handle structure for the Tx/Rx channel.
00033             (+++) Enable the DMAx interface clock.
00034             (+++) Configure the declared DMA handle structure with the required Tx/Rx parameters.
00035             (+++) Configure the DMA Tx/Rx channel.
00036             (+++) Associate the initialized DMA handle to the SMARTCARD DMA Tx/Rx handle.
00037             (+++) Configure the priority and enable the NVIC for the transfer complete interrupt on the DMA Tx/Rx channel.
00038             (+++) Configure the USARTx interrupt priority and enable the NVIC USART IRQ handle
00039                   (used for last byte sending completion detection in DMA non circular mode)

00042         flow control and Mode(Receiver/Transmitter) in the SMARTCARD Init structure.
00043

00045         (++) These APIs configure also the low level Hardware GPIO, CLOCK, CORTEX...etc)
00046              by calling the customized HAL_SMARTCARD_MspInit() API.

00125     and the Callback ID.

00129     (+) ErrorCallback             : Error Callback.
00130     (+) AbortCpltCallback         : Abort Complete Callback.
00131     (+) AbortTransmitCpltCallback : Abort Transmit Complete Callback.
00132     (+) AbortReceiveCpltCallback  : Abort Receive Complete Callback.
00133     (+) MspInitCallback           : SMARTCARD MspInit.
00134     (+) MspDeInitCallback         : SMARTCARD MspDeInit.

00233   ==============================================================================
00234               ##### Initialization and Configuration functions #####
00235   ==============================================================================
00236   [..]
00237   This subsection provides a set of functions allowing to initialize the USART
00238   in Smartcard mode.
00239   [..]
00240   The Smartcard interface is designed to support asynchronous protocol Smartcards as
00241   defined in the ISO 7816-3 standard.
00242   [..]
00243   The USART can provide a clock to the smartcard through the SCLK output.
00244   In smartcard mode, SCLK is not associated to the communication but is simply derived
00245   from the internal peripheral input clock through a 5-bit prescaler.
00246   [..]
00247   (+) For the Smartcard mode only these parameters can be configured:
00248       (++) Baud Rate
00249       (++) Word Length => Should be 9 bits (8 bits + parity)
00250       (++) Stop Bit
00251       (++) Parity: => Should be enabled
00252       (++) USART polarity
00253       (++) USART phase
00254       (++) USART LastBit
00255       (++) Receiver/transmitter modes
00256       (++) Prescaler
00257       (++) GuardTime
00258       (++) NACKState: The Smartcard NACK state
00259 
00260      (+) Recommended SmartCard interface configuration to get the Answer to Reset from the Card:
00261         (++) Word Length = 9 Bits
00262         (++) 1.5 Stop Bit
00263         (++) Even parity
00264         (++) BaudRate = 12096 baud
00265         (++) Tx and Rx enabled
00266   [..]
00267   Please refer to the ISO 7816-3 specification for more details.
00268 
00269   [..]
00270    (@) It is also possible to choose 0.5 stop bit for receiving but it is recommended
00271        to use 1.5 stop bits for both transmitting and receiving to avoid switching
00272        between the two configurations.
00273   [..]
00274     The HAL_SMARTCARD_Init() function follows the USART  SmartCard configuration
00275     procedures (details for the procedures are available in reference manuals
00276     (RM0008 for STM32F10Xxx MCUs and RM0041 for STM32F100xx MCUs)).

00278 @endverbatim

00283        |---------------------|---------------------------------------|

00333

00343   CLEAR_BIT(hsc->Instance->CR2, USART_CR2_LINEN);

00348 
00349   /* Enable the SMARTCARD Framing Error Interrupt */
00350   SET_BIT(hsc->Instance->CR3, USART_CR3_EIE);
00351 
00352   /* Enable the Peripheral */
00353   __HAL_SMARTCARD_ENABLE(hsc);
00354 
00355   /* Configure the Smartcard NACK state */

00357 
00358   /* Enable the SC mode by setting the SCEN bit in the CR3 register */

00361   /* Initialize the SMARTCARD state*/
00362   hsc->ErrorCode = HAL_SMARTCARD_ERROR_NONE;
00363   hsc->gState= HAL_SMARTCARD_STATE_READY;
00364   hsc->RxState= HAL_SMARTCARD_STATE_READY;

01522 /**
01523   * @brief This function handles SMARTCARD interrupt request.
01524   * @param  hsc    Pointer to a SMARTCARD_HandleTypeDef structure that contains
01525   *                the configuration information for SMARTCARD module.
01526   * @retval None
01527   */
01528 void HAL_SMARTCARD_IRQHandler(SMARTCARD_HandleTypeDef *hsc)
01529 {
01530   uint32_t isrflags   = READ_REG(hsc->Instance->SR);
01531   uint32_t cr1its     = READ_REG(hsc->Instance->CR1);
01532   uint32_t cr3its     = READ_REG(hsc->Instance->CR3);
01533   uint32_t dmarequest = 0x00U;
01534   uint32_t errorflags = 0x00U;
01535 
01536   /* If no error occurs */
01537   errorflags = (isrflags & (uint32_t)(USART_SR_PE | USART_SR_FE | USART_SR_ORE | USART_SR_NE));
01538   if(errorflags == RESET)
01539   {
01540     /* SMARTCARD in mode Receiver -------------------------------------------------*/
01541     if(((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
01542     {
01543       SMARTCARD_Receive_IT(hsc);
01544       return;
01545     }
01546   }
01547 
01548   /* If some errors occur */
01549   if((errorflags != RESET) && (((cr3its & USART_CR3_EIE) != RESET) || ((cr1its & (USART_CR1_RXNEIE | USART_CR1_PEIE)) != RESET)))
01550   {
01551     /* SMARTCARD parity error interrupt occurred ---------------------------*/
01552     if(((isrflags & SMARTCARD_FLAG_PE) != RESET) && ((cr1its & USART_CR1_PEIE) != RESET))
01553     {
01554       hsc->ErrorCode |= HAL_SMARTCARD_ERROR_PE;
01555     }
01556 
01557     /* SMARTCARD frame error interrupt occurred ----------------------------*/
01558     if(((isrflags & SMARTCARD_FLAG_FE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
01559     {
01560       hsc->ErrorCode |= HAL_SMARTCARD_ERROR_FE;
01561     }
01562 
01563     /* SMARTCARD noise error interrupt occurred ----------------------------*/
01564     if(((isrflags & SMARTCARD_FLAG_NE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
01565     {
01566       hsc->ErrorCode |= HAL_SMARTCARD_ERROR_NE;
01567     }
01568 
01569     /* SMARTCARD Over-Run interrupt occurred -------------------------------*/
01570     if(((isrflags & SMARTCARD_FLAG_ORE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
01571     {
01572       hsc->ErrorCode |= HAL_SMARTCARD_ERROR_ORE;
01573     }
01574     /* Call the Error call Back in case of Errors --------------------------*/
01575     if(hsc->ErrorCode != HAL_SMARTCARD_ERROR_NONE)
01576     {
01577       /* SMARTCARD in mode Receiver ----------------------------------------*/
01578       if(((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
01579       {
01580         SMARTCARD_Receive_IT(hsc);
01581       }
01582 
01583       /* If Overrun error occurs, or if any error occurs in DMA mode reception,
01584          consider error as blocking */
01585       dmarequest = HAL_IS_BIT_SET(hsc->Instance->CR3, USART_CR3_DMAR);
01586       if(((hsc->ErrorCode & HAL_SMARTCARD_ERROR_ORE) != RESET) || dmarequest)
01587       {
01588         /* Blocking error : transfer is aborted
01589           Set the SMARTCARD state ready to be able to start again the process,
01590           Disable Rx Interrupts, and disable Rx DMA request, if ongoing */
01591         SMARTCARD_EndRxTransfer(hsc);
01592         /* Disable the SMARTCARD DMA Rx request if enabled */
01593         if(HAL_IS_BIT_SET(hsc->Instance->CR3, USART_CR3_DMAR))
01594         {
01595           CLEAR_BIT(hsc->Instance->CR3, USART_CR3_DMAR);
01596 
01597           /* Abort the SMARTCARD DMA Rx channel */
01598           if(hsc->hdmarx != NULL)
01599           {
01600             /* Set the SMARTCARD DMA Abort callback :
01601               will lead to call HAL_SMARTCARD_ErrorCallback() at end of DMA abort procedure */
01602             hsc->hdmarx->XferAbortCallback = SMARTCARD_DMAAbortOnError;
01603 
01604            if(HAL_DMA_Abort_IT(hsc->hdmarx) != HAL_OK)
01605             {
01606               /* Call Directly XferAbortCallback function in case of error */
01607               hsc->hdmarx->XferAbortCallback(hsc->hdmarx);
01608             }
01609           }
01610           else
01611           {
01612 #if (USE_HAL_SMARTCARD_REGISTER_CALLBACKS == 1)
01613             /* Call registered user error callback */
01614             hsc->ErrorCallback(hsc);
01615 #else
01616             /* Call legacy weak user error callback */
01617             HAL_SMARTCARD_ErrorCallback(hsc);
01618 #endif /* USE_HAL_SMARTCARD_REGISTER_CALLBACK */
01619           }
01620         }
01621         else
01622         {
01623 #if (USE_HAL_SMARTCARD_REGISTER_CALLBACKS == 1)
01624           /* Call registered user error callback */
01625           hsc->ErrorCallback(hsc);
01626 #else
01627           /* Call legacy weak user error callback */
01628           HAL_SMARTCARD_ErrorCallback(hsc);
01629 #endif /* USE_HAL_SMARTCARD_REGISTER_CALLBACK */
01630         }
01631       }
01632       else
01633       {
01634         /* Non Blocking error : transfer could go on.
01635            Error is notified to user through user error callback */
01636 #if (USE_HAL_SMARTCARD_REGISTER_CALLBACKS == 1)
01637         /* Call registered user error callback */
01638         hsc->ErrorCallback(hsc);
01639 #else
01640         /* Call legacy weak user error callback */
01641         HAL_SMARTCARD_ErrorCallback(hsc);
01642 #endif /* USE_HAL_SMARTCARD_REGISTER_CALLBACK */
01643         hsc->ErrorCode = HAL_SMARTCARD_ERROR_NONE;
01644       }
01645     }
01646     return;
01647   } /* End if some error occurs */
01648 
01649   /* SMARTCARD in mode Transmitter ------------------------------------------*/
01650   if(((isrflags & SMARTCARD_FLAG_TXE) != RESET) && ((cr1its & USART_CR1_TXEIE) != RESET))
01651   {
01652     SMARTCARD_Transmit_IT(hsc);
01653     return;
01654   }
01655 
01656   /* SMARTCARD in mode Transmitter (transmission end) -----------------------*/
01657   if(((isrflags & SMARTCARD_FLAG_TC) != RESET) && ((cr1its & USART_CR1_TCIE) != RESET))
01658   {
01659     SMARTCARD_EndTransmit_IT(hsc);
