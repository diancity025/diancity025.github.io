00032             (+++) Enable the DMAx interface clock.

00034             (+++) Configure the DMA Tx/Rx channel.
00035             (+++) Associate the initialized DMA handle to the IRDA DMA Tx/Rx handle.
00036             (+++) Configure the priority and enable the NVIC for the transfer complete interrupt on the DMA Tx/Rx channel.
00037             (+++) Configure the IRDAx interrupt priority and enable the NVIC USART IRQ handle

00041         and Mode(Receiver/Transmitter) in the hirda Init structure.

00768 {

00770   uint32_t tickstart = 0U;
00771 
00772   /* Check that a Tx process is not already ongoing */
00773   if (hirda->gState == HAL_IRDA_STATE_READY)
00774   {
00775     if ((pData == NULL) || (Size == 0U))
00776     {
00777       return  HAL_ERROR;
00778     }

00782

00790     hirda->TxXferCount = Size;
00791     while (hirda->TxXferCount > 0U)
00792     {
00793       hirda->TxXferCount--;
00794       if (hirda->Init.WordLength == IRDA_WORDLENGTH_9B)
00795       {
00796         if (IRDA_WaitOnFlagUntilTimeout(hirda, IRDA_FLAG_TXE, RESET, tickstart, Timeout) != HAL_OK)

00798           return HAL_TIMEOUT;

00802         if (hirda->Init.Parity == IRDA_PARITY_NONE)
00803         {
00804           pData += 2U;
00805         }
00806         else
00807         {
00808           pData += 1U;
00809         }
00810       }
00811       else
00812       {

00814         {
00815           return HAL_TIMEOUT;
00816         }
00817         hirda->Instance->DR = (*pData++ & (uint8_t)0xFF);
00818       }
00819     }
00820 
00821     if (IRDA_WaitOnFlagUntilTimeout(hirda, IRDA_FLAG_TC, RESET, tickstart, Timeout) != HAL_OK)
00822     {
00823       return HAL_TIMEOUT;
00824     }
00825

01706 void HAL_IRDA_IRQHandler(IRDA_HandleTypeDef *hirda)
01707 {
01708   uint32_t isrflags   = READ_REG(hirda->Instance->SR);
01709   uint32_t cr1its     = READ_REG(hirda->Instance->CR1);
01710   uint32_t cr3its     = READ_REG(hirda->Instance->CR3);
01711   uint32_t errorflags = 0x00U;
01712   uint32_t dmarequest = 0x00U;
01713 
01714   /* If no error occurs */
01715   errorflags = (isrflags & (uint32_t)(USART_SR_PE | USART_SR_FE | USART_SR_ORE | USART_SR_NE));
01716   if (errorflags == RESET)
01717   {
01718     /* IRDA in mode Receiver -----------------------------------------------*/
01719     if (((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
01720     {
01721       IRDA_Receive_IT(hirda);
01722       return;
01723     }
01724   }
01725 
01726   /* If some errors occur */
01727   if ((errorflags != RESET) && (((cr3its & USART_CR3_EIE) != RESET) || ((cr1its & (USART_CR1_RXNEIE | USART_CR1_PEIE)) != RESET)))
01728   {
01729     /* IRDA parity error interrupt occurred -------------------------------*/
01730     if (((isrflags & USART_SR_PE) != RESET) && ((cr1its & USART_CR1_PEIE) != RESET))
01731     {
01732       hirda->ErrorCode |= HAL_IRDA_ERROR_PE;
01733     }
01734 
01735     /* IRDA noise error interrupt occurred --------------------------------*/
01736     if (((isrflags & USART_SR_NE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
01737     {
01738       hirda->ErrorCode |= HAL_IRDA_ERROR_NE;
01739     }
01740 
01741     /* IRDA frame error interrupt occurred --------------------------------*/
01742     if (((isrflags & USART_SR_FE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
01743     {
01744       hirda->ErrorCode |= HAL_IRDA_ERROR_FE;
01745     }
01746 
01747     /* IRDA Over-Run interrupt occurred -----------------------------------*/
01748     if (((isrflags & USART_SR_ORE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
01749     {
01750       hirda->ErrorCode |= HAL_IRDA_ERROR_ORE;
01751     }
01752     /* Call IRDA Error Call back function if need be -----------------------*/
01753     if (hirda->ErrorCode != HAL_IRDA_ERROR_NONE)
01754     {
01755       /* IRDA in mode Receiver ---------------------------------------------*/
01756       if (((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
01757       {
01758         IRDA_Receive_IT(hirda);
01759       }
01760 
01761       /* If Overrun error occurs, or if any error occurs in DMA mode reception,
01762          consider error as blocking */
01763       dmarequest = HAL_IS_BIT_SET(hirda->Instance->CR3, USART_CR3_DMAR);
01764       if (((hirda->ErrorCode & HAL_IRDA_ERROR_ORE) != RESET) || dmarequest)
01765       {
01766         /* Blocking error : transfer is aborted
01767            Set the IRDA state ready to be able to start again the process,
01768            Disable Rx Interrupts, and disable Rx DMA request, if ongoing */
01769         IRDA_EndRxTransfer(hirda);
01770 
01771         /* Disable the IRDA DMA Rx request if enabled */
01772         if (HAL_IS_BIT_SET(hirda->Instance->CR3, USART_CR3_DMAR))
01773         {
01774           CLEAR_BIT(hirda->Instance->CR3, USART_CR3_DMAR);
01775 
01776           /* Abort the IRDA DMA Rx channel */
01777           if (hirda->hdmarx != NULL)
01778           {
01779             /* Set the IRDA DMA Abort callback :
01780             will lead to call HAL_IRDA_ErrorCallback() at end of DMA abort procedure */
01781             hirda->hdmarx->XferAbortCallback = IRDA_DMAAbortOnError;
01782 
01783             /* Abort DMA RX */
01784             if (HAL_DMA_Abort_IT(hirda->hdmarx) != HAL_OK)
01785             {
01786               /* Call Directly XferAbortCallback function in case of error */
01787               hirda->hdmarx->XferAbortCallback(hirda->hdmarx);
01788             }
01789           }
01790           else
01791           {
01792 #if (USE_HAL_IRDA_REGISTER_CALLBACKS == 1)
01793             /* Call registered user error callback */
01794             hirda->ErrorCallback(hirda);
01795 #else
01796             /* Call legacy weak user error callback */
01797             HAL_IRDA_ErrorCallback(hirda);
01798 #endif /* USE_HAL_IRDA_REGISTER_CALLBACK */
01799           }
01800         }
01801         else
01802         {
01803 #if (USE_HAL_IRDA_REGISTER_CALLBACKS == 1)
01804           /* Call registered user error callback */
01805           hirda->ErrorCallback(hirda);
01806 #else
01807           /* Call legacy weak user error callback */
01808           HAL_IRDA_ErrorCallback(hirda);
01809 #endif /* USE_HAL_IRDA_REGISTER_CALLBACK */
01810         }
01811       }
01812       else
01813       {
01814         /* Non Blocking error : transfer could go on.
01815            Error is notified to user through user error callback */
01816 #if (USE_HAL_IRDA_REGISTER_CALLBACKS == 1)
01817         /* Call registered user error callback */
01818         hirda->ErrorCallback(hirda);
01819 #else
01820         /* Call legacy weak user error callback */
01821         HAL_IRDA_ErrorCallback(hirda);
01822 #endif /* USE_HAL_IRDA_REGISTER_CALLBACK */
01823 
01824         hirda->ErrorCode = HAL_IRDA_ERROR_NONE;
01825       }
01826     }
01827     return;
01828   } /* End if some error occurs */
01829 
01830   /* IRDA in mode Transmitter ------------------------------------------------*/
01831   if (((isrflags & USART_SR_TXE) != RESET) && ((cr1its & USART_CR1_TXEIE) != RESET))
01832   {
01833     IRDA_Transmit_IT(hirda);
01834     return;
01835   }
01836 
01837   /* IRDA in mode Transmitter end --------------------------------------------*/
01838   if (((isrflags & USART_SR_TC) != RESET) && ((cr1its & USART_CR1_TCIE) != RESET))
01839   {
01840     IRDA_EndTransmit_IT(hirda);
01841     return;
01842   }
01843 }
